# aegis v0.1 — Passive Vulnerability Coverage — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A new `aegis` plugin that turns a site's fingerprint (from beacon) into a prioritized, passive vulnerability-coverage report — known CVEs (OSV-first) ranked by CISA KEV + EPSS, plus TLS/header misconfig.

**Architecture:** A one-shot passive pipeline of small stdlib (`urllib`) clients chained by an orchestrator: fingerprint → OSV/NVD CVE lookup → KEV/EPSS overlay → SSL-Labs/Observatory misconfig → `coverage.{md,json}`. No active scanning; no third-party deps.

**Tech Stack:** Python 3 stdlib only (`urllib.request`, `json`, `argparse`, `unittest.mock`); pytest; Claude Code plugin skill/command.

**Spec:** `docs/superpowers/specs/2026-07-15-aegis-vulnerability-coverage-design.md`

## Global Constraints

- **Stdlib only.** No `httpx`/`requests`/`fire`/PyYAML. All HTTP via `urllib.request` through the shared `_http.py`.
- **Fail-safe, always.** Any network error / non-2xx / bad JSON → the client returns `{"error": "<reason>"}` (never raises); the orchestrator records that source in `summary.coverage_incomplete`. Never a crash, never a silent "0 vulnerabilities."
- **Honesty signals:** a component with no version → `version_known: false` + `[VERSION-UNKNOWN:{name}]`; a failed feed → listed in `coverage_incomplete`. Gaps are surfaced, never read as "clean."
- **OSV-first:** packages → OSV `(ecosystem,name,version)`; frameworks/servers → NVD keyword search `"{name} {version}"` (no CPE construction). KEV+EPSS are prioritization overlays only.
- **Report ordering:** KEV first, then CVSS≥7 & EPSS≥0.1, then the rest.
- **Paths:** report → `docs/sites/{slug}/security/coverage.{md,json}`, resolved from repo-root cwd. Slug derivation reuses beacon's rule (mirror `plugins/beacon/skills/site-recon/scripts/slugify.py` — call it, don't re-implement).
- **Version:** `aegis` ships at `0.1.0`. Passive only — no nuclei/active probing (that's v0.2).
- **Every client is both a CLI (`python3 X.py … --json`) and importable** (functions return dict/list).

```python
#!/usr/bin/env python3
"""Shared stdlib HTTP for aegis clients: fail-safe GET/POST JSON with an on-disk
TTL cache and a courtesy rate-limit. Never raises — every failure returns
{"error": <reason>} so the orchestrator can record 'coverage incomplete' rather
than crash or report a false 'no vulnerabilities'."""
import hashlib
import json
import os
import time
from urllib import request, error

CACHE_DIR = os.path.join(".aegis-cache")
_UA = "aegis-vuln-coverage/0.1 (+https://github.com/neotherapper/claude-plugins)"
_last_call = [0.0]
_MIN_INTERVAL = 0.25  # ~4 req/s courtesy cap


def _cache_path(key):
    h = hashlib.sha256(key.encode()).hexdigest()[:32]
    return os.path.join(CACHE_DIR, h + ".json")


def _cache_get(key, ttl):
    if ttl <= 0:
        return None
    p = _cache_path(key)
    try:
        if os.path.isfile(p) and (time.time() - os.path.getmtime(p)) < ttl:
            with open(p, encoding="utf-8") as f:
                return json.load(f)
    except (OSError, ValueError):
        return None
    return None


def _cache_put(key, value):
    try:
        os.makedirs(CACHE_DIR, exist_ok=True)
        with open(_cache_path(key), "w", encoding="utf-8") as f:
            json.dump(value, f)
    except OSError:
        pass


def _throttle():
    dt = time.time() - _last_call[0]
    if dt < _MIN_INTERVAL:
        time.sleep(_MIN_INTERVAL - dt)
    _last_call[0] = time.time()


def _do(req, timeout):
    try:
        _throttle()
        with request.urlopen(req, timeout=timeout) as resp:
            body = resp.read().decode("utf-8", "replace")
        return json.loads(body)
    except (error.URLError, error.HTTPError) as e:
        return {"error": f"request failed: {e}"}
    except ValueError as e:
        return {"error": f"bad json: {e}"}
    except Exception as e:  # never raise from an advisory client
        return {"error": f"unexpected: {e}"}


def get_json(url, headers=None, timeout=20, cache_ttl=86400):
    cached = _cache_get(url, cache_ttl)
    if cached is not None:
        return cached
    h = {"Accept": "application/json", "User-Agent": _UA, **(headers or {})}
    result = _do(request.Request(url, headers=h), timeout)
    if "error" not in result:
        _cache_put(url, result)
    return result


def post_json(url, payload, headers=None, timeout=20):
    h = {"Accept": "application/json", "Content-Type": "application/json",
         "User-Agent": _UA, **(headers or {})}
    data = json.dumps(payload).encode()
    return _do(request.Request(url, data=data, headers=h, method="POST"), timeout)
```

```python
#!/usr/bin/env python3
"""OSV.dev query — packages/JS-libs -> known vulns (no auth). Primary CVE source."""
import argparse, json, sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).parent))
import _http

OSV_QUERY = "https://api.osv.dev/v1/query"

def _cvss(sev):
    scores = [float(s["score"]) for s in (sev or []) if s.get("score", "").replace(".", "").isdigit()]
    return max(scores) if scores else None

def query(name, version, ecosystem="npm"):
    r = _http.post_json(OSV_QUERY, {"version": version,
                                    "package": {"name": name, "ecosystem": ecosystem}})
    if not isinstance(r, dict) or "vulns" not in r:
        return []
    out = []
    for v in r["vulns"]:
        out.append({"id": v.get("id"), "summary": v.get("summary") or v.get("details", "")[:200],
                    "cvss": _cvss(v.get("severity")), "aliases": v.get("aliases", []), "source": "osv"})
    return out

def main(argv=None):
    p = argparse.ArgumentParser()
    p.add_argument("--name", required=True); p.add_argument("--version", required=True)
    p.add_argument("--ecosystem", default="npm"); p.add_argument("--json", action="store_true")
    a = p.parse_args(argv)
    print(json.dumps(query(a.name, a.version, a.ecosystem), indent=2))
    return 0

if __name__ == "__main__":
    sys.exit(main())
```
