```python
#!/usr/bin/env python3
"""
har-reconstruct.py — Convert Chrome DevTools MCP or cmux network capture JSON to HAR 1.2

Usage:
  python3 har-reconstruct.py --input .beacon/chrome-requests.json \
                              --output .beacon/capture.har \
                              --domain example.com

Auto-detects input format:
  Chrome MCP: entries have a numeric "reqid" field
  cmux:       entries have a "request" object with "url" and "method"
"""

import argparse
import json
import sys
from datetime import datetime, timezone
from urllib.parse import urlparse

ANALYTICS_DOMAINS = {
    "google-analytics.com",
    "googletagmanager.com",
    "gtm.js",
    "mixpanel.com",
    "hotjar.com",
    "amplitude.com",
    "segment.io",
    "segment.com",
    "facebook.net",
    "doubleclick.net",
    "ads.twitter.com",
}

SKIP_EXTENSIONS = {".woff", ".woff2", ".ttf", ".eot", ".png", ".jpg", ".jpeg",
                   ".gif", ".svg", ".ico", ".webp", ".mp4", ".mp3"}

API_CONTENT_TYPES = {
    "application/json",
    "text/plain",
    "application/x-www-form-urlencoded",
    "text/html",  # keep — some APIs return HTML fragments
    "application/xml",
    "text/xml",
}


def is_analytics(url: str) -> bool:
    host = urlparse(url).netloc.lower()
    return any(ad in host for ad in ANALYTICS_DOMAINS)


def is_static_asset(url: str) -> bool:
    path = urlparse(url).path.lower()
    return any(path.endswith(ext) for ext in SKIP_EXTENSIONS)


def should_keep(url: str, domain: str) -> bool:
    if not url:
        return False
    if domain and domain not in urlparse(url).netloc:
        return False
    if is_analytics(url):
        return False
    if is_static_asset(url):
        return False
    return True


def headers_dict_to_list(headers: dict) -> list:
    if not headers:
        return []
    return [{"name": k, "value": str(v)} for k, v in headers.items()]


def make_har_entry(method: str, url: str, status: int,
                   req_headers: dict, resp_headers: dict,
                   resp_body: str) -> dict:
    content_type = "application/json"
    for k, v in (resp_headers or {}).items():
        if k.lower() == "content-type":
            content_type = v.split(";")[0].strip()
            break

    return {
        "startedDateTime": datetime.now(timezone.utc).isoformat(),
        "time": 100,
        "cache": {},
        "timings": {"send": 0, "wait": 100, "receive": 0},
        "request": {
            "method": method.upper(),
            "url": url,
            "httpVersion": "HTTP/1.1",
            "headers": headers_dict_to_list(req_headers),
            "cookies": [],
            "queryString": [],
            "headersSize": -1,
            "bodySize": -1,
        },
        "response": {
            "status": status,
            "statusText": "",
            "httpVersion": "HTTP/1.1",
            "headers": headers_dict_to_list(resp_headers),
            "cookies": [],
            "redirectURL": "",
            "headersSize": -1,
            "bodySize": -1,
            "content": {
                "size": len(resp_body) if resp_body else 0,
                "mimeType": content_type,
                "text": resp_body or "",
            },
        },
    }


def parse_chrome_mcp(entries: list, domain: str) -> list:
    """Parse Chrome DevTools MCP list_network_requests output (has 'reqid' field)."""
    har_entries = []
    for e in entries:
        url = e.get("url", "")
        if not should_keep(url, domain):
            continue
        har_entries.append(make_har_entry(
            method=e.get("method", "GET"),
            url=url,
            status=e.get("status", 0),
            req_headers=e.get("request_headers", {}),
            resp_headers=e.get("response_headers", {}),
            resp_body=e.get("response_body", ""),
        ))
    return har_entries


def parse_cmux(entries: list, domain: str) -> list:
    """Parse cmux browser network requests output."""
    har_entries = []
    for e in entries:
        # cmux format: {"request": {"url": ..., "method": ...}, "response": {...}}
        req = e.get("request", e)  # fallback: entry is flat
        resp = e.get("response", {})
        url = req.get("url", e.get("url", ""))
        if not should_keep(url, domain):
            continue
        har_entries.append(make_har_entry(
            method=req.get("method", e.get("method", "GET")),
            url=url,
            status=resp.get("status", e.get("status", 0)),
            req_headers=req.get("headers", e.get("request_headers", {})),
            resp_headers=resp.get("headers", e.get("response_headers", {})),
            resp_body=resp.get("body", e.get("response_body", "")),
        ))
    return har_entries


def reconstruct(input_path: str, output_path: str, domain: str) -> None:
    with open(input_path) as f:
        data = json.load(f)

    if not isinstance(data, list):
        print(f"[ERROR] Expected JSON array, got {type(data).__name__}", file=sys.stderr)
        sys.exit(1)

    # Auto-detect format: Chrome MCP entries have numeric "reqid"
    is_chrome = data and isinstance(data[0].get("reqid"), (int, float))

    if is_chrome:
        entries = parse_chrome_mcp(data, domain)
    else:
        entries = parse_cmux(data, domain)

    har = {
        "log": {
            "version": "1.2",
            "creator": {"name": "beacon-plugin", "version": "0.2.0"},
            "entries": entries,
        }
    }

    with open(output_path, "w") as f:
        json.dump(har, f, indent=2)

    print(f"[har-reconstruct] {len(entries)} entries written to {output_path}")


def main():
    parser = argparse.ArgumentParser(
        description="Convert Chrome MCP or cmux network capture JSON to HAR 1.2"
    )
    parser.add_argument("--input", required=True,
                        help="Path to chrome-requests.json or cmux-requests.json")
    parser.add_argument("--output", required=True,
                        help="Output HAR file path")
    parser.add_argument("--domain", default="",
                        help="Target domain to keep (e.g. example.com). Filters out other domains.")
    args = parser.parse_args()

    reconstruct(args.input, args.output, args.domain)


if __name__ == "__main__":
    main()
```

# IMPORTANT: Corrected Chrome DevTools MCP v0.21.0 signatures
# new_page has no required args and returns {page_id: "..."}
# navigate_page takes (url, type="url") — NOT (url, page_id)

3. navigate_page(url, type="url", timeout=10000)
4. Poll readyState:
   evaluate_script(() => document.readyState)
   → If not "complete", retry up to 3× with 2s delay

5. Extract JS globals:
   evaluate_script(() => {
     const globals = {};
     // Framework-specific (from tech pack Section 3):
     ['__NEXT_DATA__','__NUXT__','Shopify','__remixContext','__initialData__'].forEach(k => {
       if (window[k]) globals[k] = window[k];
     });
     // Catch-all __ prefixed globals:
     Object.keys(window).filter(k => k.startsWith('__')).forEach(k => {
       if (!globals[k]) globals[k] = typeof window[k];
     });
     return JSON.stringify(globals);
   })

6. Capture API network traffic:
   list_network_requests({resourceTypes: ["xhr", "fetch"]})
   → Filter client-side: keep only URLs containing target domain
   → For each matching request: get_network_request(reqid) — captures response body

7. Execute browse plan actions (click, fill, etc.) if specified for this URL

8. After interactions: list_network_requests({resourceTypes: ["xhr", "fetch"]}) again

9. take_snapshot() — accessibility tree for auth-gate detection
```
